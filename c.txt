#include<stdio.h>
#include<stdlib.h>
int main() {
 	int marks,total=0,counter; 
float avg; 
for(counter=1;counter<=10;counter++)
 { 
printf("Enter the marks for subject %d : ",counter);
 	scanf("%d",&marks); 
total=total+marks;
 }
 avg=(float)total/10;
 printf("Total of Marks : %d\n",total);
 printf("Average of Marks : %.2f ",avg);

if(avg<50)
printf("Fail\n");
 	else
 		printf("Pass\n");
 return 0; 
}
